335 int AttemptToDecodePacket(double sinr, double capture_effect, double pd,
336  double power_rx_interest, double constant_per, int node_id, int packet_type,
337  int destination_id){
338 
339  int packet_lost;
340  double per (0);
341 
342  // Try to decode when power received is greater than pd
343  if(sinr < capture_effect || power_rx_interest < pd) {
344  per = 1;
345  } else {
346  // Just apply PER to DATA packets
347  if( (destination_id == node_id) && (packet_type == PACKET_TYPE_DATA) ){
348  per = constant_per;
349  }
350  }
351 
352  packet_lost = ((double) rand() / (RAND_MAX)) < per;
353 
354  return packet_lost;
355 }

371 int IsPacketLost(int primary_channel, Notification incoming_notification, Notification new_notification,
372  double sinr, double capture_effect, double pd, double power_rx_interest, double constant_per,
373  int node_id, int capture_effect_model){
374 
375  int loss_reason (PACKET_NOT_LOST);
376  int is_packet_lost; // Determines if the current notification has been lost (1) or not (0)
377 
378  switch(capture_effect_model) {
379 
380  case CE_DEFAULT: {
381 
382  // Sergio on 25 Oct 2017:
383  // - Change the way packets are determined are lost
384  // - Use both incoming (interest) and new (sometimes noisy) notifications
385  // - We were missing some cases. E.g. when RX_DATA and new packet arrived
386 
387  // Check if incoming notification (of interest) involves the primary channel
388  if(primary_channel >= incoming_notification.left_channel && primary_channel <= incoming_notification.right_channel){
389 
390  // Attempt to decode (or continue decoding) the notification of interest
391  is_packet_lost = AttemptToDecodePacket(sinr, capture_effect, pd, power_rx_interest, constant_per, node_id,
392  new_notification.packet_type, new_notification.destination_id);
393 
394  if (is_packet_lost) { // Incoming packet is lost
395  if (power_rx_interest < pd) { // Signal strength is not enough (< pd) to be decoded
396  loss_reason = PACKET_LOST_LOW_SIGNAL;
397 // hidden_nodes_list[new_notification.source_id] = TRUE;
398  } else if (sinr < capture_effect){ // Capture effect not accomplished
399  loss_reason = PACKET_LOST_INTERFERENCE;
400  } else { // Incoming packet lost due to PER
401  loss_reason = PACKET_LOST_SINR_PROB;
402  }
403  }
404 
405  } else{
406 
407  loss_reason = PACKET_LOST_OUTSIDE_CH_RANGE;
408 
409  }
410 
411  break;
412  }
413 
414  case CE_IEEE_802_11: {
415 
416  // Check if the RSSI is higher than the CST
417  if (power_rx_interest > pd) {
418  // The packet can be properly decoded
419  loss_reason = -1;
420  } else {
421  loss_reason = PACKET_LOST_LOW_SIGNAL;
422  }
423 
424  break;
425  }
426 
427  }
428 
429  return loss_reason;
430 
431 }
