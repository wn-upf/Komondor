71 double GenerateReward(int type_of_reward, Performance performance) {
72 
73  double reward;
74 
75  // Switch to select the reward according to the metric used (rewards must be normalized)
76  switch(type_of_reward){
77 
78  /* PERFORMANCE_PACKETS_SENT:
79  * - The number of packets sent are taken into account
80  * - The reward must be bounded by the maximum number of data packets
81  * that can be sent in each interval (e.g., packets that were sent but lost)
82  */
83  case REWARD_TYPE_PACKETS_SENT:{
84  reward = performance.data_packets_sent/performance.data_packets_lost;
85  break;
86  }
87 
88  /* PERFORMANCE_THROUGHPUT:
89  * - The throughput experienced during the last period is taken into account
90  * - The reward must be bounded by the maximum throughput that would be experienced
91  * (e.g., consider the data rate granted by the modulation and the total time)
92  */
93  case REWARD_TYPE_THROUGHPUT:{
94 
95  if (performance.max_bound_throughput == 0) {
96  reward = 0;
97  } else {
98  reward = performance.throughput/performance.max_bound_throughput;
99  }
100  break;
101  }
102 
103  /* REWARD_TYPE_PACKETS_GENERATED:
104  * -
105  */
106  case REWARD_TYPE_PACKETS_GENERATED:{
107  reward = (performance.num_packets_generated - performance.num_packets_dropped) /
108  performance.num_packets_generated;
109  break;
110  }
111 
112  default:{
113  printf("ERROR: %d is not a correct type of performance indicator\n", type_of_reward);
114  exit(EXIT_FAILURE);
115  break;
116  }
117 
118  }
119 
120  return reward;
121 
122 }
